const request = require('request')

const geocode = (address, callback) => {
    const url = 'https://api.mapbox.com/geocoding/v5/mapbox.places/' + encodeURIComponent(address) + '.json?access_token=pk.eyJ1Ijoiam9uYWNvbnRyZXJhcyIsImEiOiJja3NzZDV5eDUwZW5xMnVwaDJqOXhjdG4wIn0.SXIoONRGTmEA996_U99C1A&limit=1'

    request({ url, json: true }, (error, { body }) => {
        if (error) {
            callback('Unable to connect to location services.', undefined)
        } else if (body.features.length == 0) {
            callback('Unable to find location. Try another search', undefined)
        } else {
            let latitude = body.features[0].center[1];
            let longitude = body.features[0].center[0];
            let location = body.features[0].place_name;

            callback(undefined, {
                latitude: latitude,
                longitude: longitude,
                location: location
            })
        }
    })
}

module.exports = geocode;
